CREATE TABLE IF NOT EXISTS public.cliente
(
    codcliente integer NOT NULL DEFAULT nextval('cliente_codcliente_seq'::regclass),
    nomecliente character varying(100) COLLATE pg_catalog."default",
    rgcliente character varying(20) COLLATE pg_catalog."default",
    enderecocliente character varying(255) COLLATE pg_catalog."default",
    bairrocliente character varying(100) COLLATE pg_catalog."default",
    cidadecliente character varying(100) COLLATE pg_catalog."default",
    estadocliente character(2) COLLATE pg_catalog."default",
    cepcliente character varying(9) COLLATE pg_catalog."default",
    nascimentocliente date,
    CONSTRAINT cliente_pkey PRIMARY KEY (codcliente)
)

CREATE TABLE IF NOT EXISTS public.telefone
(
    telefone integer NOT NULL,
    codcliente integer,
    CONSTRAINT telefone_pkey PRIMARY KEY (telefone),
    CONSTRAINT telefone_codcliente_fkey FOREIGN KEY (codcliente)
        REFERENCES public.cliente (codcliente) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)

CREATE TABLE IF NOT EXISTS public.chale
(
    codchale integer NOT NULL DEFAULT nextval('chale_codchale_seq'::regclass),
    localizacao character varying(100) COLLATE pg_catalog."default",
    capacidade integer,
    valoraltaestacao double precision,
    valorbaixaestacao double precision,
    CONSTRAINT chale_pkey PRIMARY KEY (codchale)
)

CREATE TABLE IF NOT EXISTS public.hospedagem
(
    codhospedagem integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    codcliente integer,
    codchale integer,
    datainicio date,
    datafim date,
    desconto integer,
    estado character varying(50) COLLATE pg_catalog."default",
    qtdpessoas integer,
    valorfinal double precision,
    CONSTRAINT hospedagem_pkey PRIMARY KEY (codhospedagem)
)
